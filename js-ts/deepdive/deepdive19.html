                                                                                              нҒҙлһҳмҠӨ(class)

                                                                                              ES6м—җм„ң нҒҙлһҳмҠӨк°Җ лҸ„мһ…лҗҳм—ҲлӢӨ. н•ҳм§Җл§Ң ES6мқҳ нҒҙлһҳмҠӨк°Җ кё°мЎҙ н”„лЎңнҶ нғҖмһ… кё°л°ҳ к°қмІҙм§Җн–Ҙ лӘЁлҚёмқ„ нҸҗм§Җн•ҳкі  мғҲлЎңмҡҙ к°қмІҙм§Җн–Ҙ лӘЁлҚёмқ„ м ңкіөн•ҳлҠ” кІғмқҖ м•„лӢҲлӢӨ. мӮ¬мӢӨ нҒҙлһҳмҠӨлҸ„ н•ЁмҲҳмқҙл©°, кё°мЎҙ н”„лЎңнҶ нғҖмһ… кё°л°ҳ нҢЁн„ҙмқҳ л¬ёлІ•м Ғ м„Өнғ•мқҙлқјкі  ліј мҲҳ мһҲлӢӨ.

                                                                                              нҒҙлһҳмҠӨмҷҖ мғқм„ұмһҗ н•ЁмҲҳлҠ” лӘЁл‘җ н”„лЎңнҶ нғҖмһ… кё°л°ҳмқҳ мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•ҳм§Җл§Ң м •нҷ•нһҲ лҸҷмқјн•ҳкІҢ лҸҷмһ‘н•ҳм§ҖлҠ” м•ҠлҠ”лӢӨ.
                                                                                              нҒҙлһҳмҠӨлҠ” мғқм„ұмһҗ н•ЁмҲҳліҙлӢӨ м—„кІ©н•ҳл©° нҒҙлһҳмҠӨлҠ” мғқм„ұмһҗ н•ЁмҲҳм—җм„ңлҠ” м ңкіөн•ҳм§Җ м•ҠлҠ” кё°лҠҘлҸ„ м ңкіөн•ңлӢӨ.

                                                                                              л”°лқјм„ң нҒҙлһҳмҠӨлҘј н”„лЎңнҶ нғҖмһ… кё°л°ҳ к°қмІҙ мғқм„ұнҢЁн„ҙмқҳ лӢЁмҲңн•ң л¬ёлІ•м Ғ м„Өнғ•мңјлЎң ліҙкё°ліҙлӢӨлҠ” мғҲлЎңмҡҙ к°қмІҙ мғқм„ұ л©”м»ӨлӢҲмҰҳмңјлЎң ліҙлҠ”кІғмқҙ мўҖ лҚ” н•©лӢ№н•ҳлӢӨкі  н•  мҲҳ мһҲлӢӨ.

                                                                                              нҒҙлһҳмҠӨм—җ лҢҖн•ҙм„ңлҠ” 25мһҘ "нҒҙлһҳмҠӨ"м—җм„ң мһҗм„ёнһҲ мӮҙнҺҙліҙмһҗ.

                                                                                              мһҗл°”мҠӨнҒ¬лҰҪнҠёлҠ” к°қмІҙ кё°л°ҳмқҳ н”„лЎңк·ёлһҳл°Қ м–ём–ҙмқҙл©° мһҗл°”мҠӨнҒ¬лҰҪнҠёлҘј мқҙлЈЁкі  мһҲлҠ” кұ°мқҳ "лӘЁл“  кІғ"мқҙ к°қмІҙлӢӨ.
                                                                                              мӣҗмӢң нғҖмһ…мқҳ к°’мқ„ м ңмҷён•ң лӮҳлЁём§Җ к°’л“Ө(н•ЁмҲҳ, л°°м—ҙ, м •к·ң н‘ңнҳ„мӢқ л“ұ)мқҖ лӘЁл‘җ к°қмІҙлӢӨ.

                                                                                              # 1. к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°Қ

                                                                                              к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°ҚмқҖ н”„лЎңк·ёлһЁмқ„ лӘ…л №м–ҙ лҳҗлҠ” н•ЁмҲҳмқҳ лӘ©лЎқмңјлЎң ліҙлҠ” м „нҶөм Ғмқё лӘ…л №нҳ• н”„лЎңк·ёлһҳл°Қмқҳ м Ҳмһҗм§Җн–Ҙм Ғ кҙҖм җм—җм„ң лІ—м–ҙлӮҳ м—¬лҹ¬ к°ңмқҳ лҸ…лҰҪм Ғ лӢЁмң„, мҰү к°қмІҙмқҳ м§‘н•©мңјлЎң н”„лЎңк·ёлһЁмқ„ н‘ңнҳ„н•ҳл ӨлҠ” н”„лЎңк·ёлһҳл°Қ нҢЁлҹ¬лӢӨмһ„мқ„ л§җн•ңлӢӨ.

                                                                                              к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°ҚмқҖ мӢӨм„ёкі„мқҳ мӢӨмІҙ(мӮ¬л¬јмқҙлӮҳ к°ңл…җ)лҘј мқёмӢқн•ҳлҠ” мІ н•ҷм Ғ мӮ¬кі лҘј н”„лЎңк·ёлһҳл°Қм—җ м ‘лӘ©н•ҳл ӨлҠ” мӢңлҸ„м—җм„ң мӢңмһ‘н•ңлӢӨ. мӢӨмІҙлҠ” нҠ№м§•мқҙлӮҳ м„ұм§Ҳмқ„ лӮҳнғҖлӮҙлҠ” мҶҚм„ұ(attribute/property)мқ„ к°Җм§Җкі  мһҲкі , мқҙлҘј нҶөн•ҙ мӢӨмІҙлҘј мқёмӢқн•ҳкұ°лӮҳ кө¬лі„н•  мҲҳ мһҲлӢӨ.

                                                                                              мҳҲлҘј л“Өм–ҙ, мӮ¬лһҢмқҖ мқҙлҰ„, мЈјмҶҢ, лӮҳмқҙ, м§Ғм—… л“ұ лӢӨм–‘н•ң мҶҚм„ұмқ„ к°–лҠ”лӢӨ.
                                                                                              мқҙл•Ң "мқҙлҰ„мқҙ м•„л¬ҙк°ңмқҙкі  м„ұлі„мқҖ м—¬м„ұмқҙл©° лӮҳмқҙлҠ” 20м„ёмқё мӮ¬лһҢ"кіј к°ҷмқҙ мҶҚм„ұмқ„ кө¬мІҙм ҒмңјлЎң н‘ңнҳ„н•ҳл©ҙ нҠ№м •н•ң мӮ¬лһҢмқ„ лӢӨлҘё мӮ¬лһҢкіј кө¬лі„н•ҳм—¬ мқёмӢқн•  мҲҳ мһҲлӢӨ.

                                                                                              мқҙлҹ¬н•ң л°©мӢқмқ„ н”„лЎңк·ёлһҳл°Қм—җ м ‘лӘ©мӢңмјңліҙмһҗ. мӮ¬лһҢм—җкІҢлҠ” лӢӨм–‘н•ң мҶҚм„ұмқҙ мһҲмңјлӮҳ мҡ°лҰ¬к°Җ кө¬нҳ„н•ҳл ӨлҠ” н”„лЎңк·ёлһЁм—җм„ңлҠ” мӮ¬лһҢмқҳ "мқҙлҰ„"кіј "мЈјмҶҢ"лқјлҠ” мҶҚм„ұм—җл§Ң кҙҖмӢ¬мқҙ мһҲлӢӨкі  к°Җм •н•ҳмһҗ. мқҙмІҳлҹј лӢӨм–‘н•ң мҶҚм„ұ мӨ‘м—җм„ң н”„лЎңк·ёлһЁм—җ н•„мҡ”н•ң мҶҚм„ұл§Ң к°„м¶”л Ө лӮҙм–ҙ н‘ңнҳ„н•ҳлҠ” кІғмқ„ м¶”мғҒнҷ”(abstraction)лқј н•ңлӢӨ.

                                                                                              "мқҙлҰ„"кіј "мЈјмҶҢ"лқјлҠ” мҶҚм„ұмқ„ к°–лҠ” personмқҙлқјлҠ” к°қмІҙлҘј мһҗл°”мҠӨнҒ¬лҰҪнҠёлЎң н‘ңнҳ„н•ҳл©ҙ лӢӨмқҢкіј к°ҷлӢӨ.

                                                                                              // мқҙлҰ„кіј мЈјмҶҢ мҶҚм„ұмқ„ к°–лҠ” к°қмІҙ
const person = {
  name: "Lee",
  address: "Seoul",
};

console.log(person); // {name: "Lee", address: "Seoul"}

                                                                                              мқҙл•Ң н”„лЎңк·ёлһҳлЁё(subject, мЈјмІҙ)лҠ” мқҙлҰ„кіј мЈјмҶҢ мҶҚм„ұмңјлЎң н‘ңнҳ„лҗң к°қмІҙ(object)мқё personмқ„ лӢӨлҘё к°қмІҙмҷҖ кө¬лі„н•ҳм—¬ мқёмӢқн•  мҲҳ мһҲлӢӨ. мқҙмІҳлҹј мҶҚм„ұмқ„ нҶөн•ҙ м—¬лҹ¬ к°ңмқҳ к°’мқ„ н•ҳлӮҳмқҳ лӢЁмң„лЎң кө¬м„ұн•ң ліөн•©м Ғмқё мһҗлЈҢкө¬мЎ°лҘј к°қмІҙлқј н•ҳл©°, к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°ҚмқҖ лҸ…лҰҪм Ғмқё к°қмІҙмқҳ м§‘н•©мңјлЎң н”„лЎңк·ёлһЁмқ„ н‘ңнҳ„н•ҳл ӨлҠ” н”„лЎңк·ёлһҳл°Қ нҢЁлҹ¬лӢӨмһ„мқҙлӢӨ.

                                                                                              мқҙлІҲм—җлҠ” мӣҗ(Circle)мқҙлқјлҠ” к°ңл…җмқ„ к°қмІҙлЎң л§Ңл“Өм–ҙліҙмһҗ. мӣҗм—җлҠ” л°ҳм§ҖлҰ„мқҙлқјлҠ” мҶҚм„ұмқҙ мһҲлӢӨ. мқҙ л°ҳм§ҖлҰ„мқ„ к°Җм§Җкі  мӣҗмқҳ м§ҖлҰ„, л‘ҳл Ҳ, л„“мқҙлҘј кө¬н•  мҲҳ мһҲлӢӨ. мқҙл•Ң л°ҳм§ҖлҰ„мқҖ мӣҗмқҳ мғҒнғңлҘј лӮҳнғҖлӮҙлҠ” лҚ°мқҙн„°мқҙл©° мӣҗмқҳ м§ҖлҰ„, л‘ҳл Ҳ, л„“мқҙлҘј кө¬н•ҳлҠ” кІғмқҖ лҸҷмһ‘мқҙлӢӨ.

                                                                                              const circle = {
  radius: 5, // л°ҳм§ҖлҰ„

  // мӣҗмқҳ м§ҖлҰ„: 2r
  getDiameter() {
    return 2 * this.radius;
  },

  // мӣҗмқҳ л‘ҳл Ҳ: 2ПҖr
  getPerimeter() {
    return 2 * Math.PI * this.radius;
  },

  // мӣҗмқҳ л„“мқҙ: ПҖrr
  getArea() {
    return Math.PI * this.radius ** 2;
  },
};

console.log(circle);
// {radius: 5, getDiameter: Ж’, getPerimeter: Ж’, getArea: Ж’}

console.log(circle.getDiameter()); // 10
console.log(circle.getPerimeter()); // 31.41592653589793
console.log(circle.getArea()); // 78.53981633974483

                                                                                              мқҙмІҳлҹј к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°ҚмқҖ к°қмІҙмқҳ мғҒнғңлҘј лӮҳнғҖлӮҙлҠ” лҚ°мқҙн„°мҷҖ мғҒнғң лҚ°мқҙн„°лҘј мЎ°мһ‘н•  мҲҳ мһҲлҠ” лҸҷмһ‘мқ„ н•ҳлӮҳмқҳ л…јлҰ¬м Ғмқё лӢЁмң„лЎң л¬¶м–ҙ мғқк°Ғн•ңлӢӨ. л”°лқјм„ң к°қмІҙлҠ” мғҒнғң лҚ°мқҙн„°мҷҖ лҸҷмһ‘мқ„ н•ҳлӮҳмқҳ л…јлҰ¬м Ғмқё лӢЁмң„лЎң л¬¶мқҖ ліөн•©м Ғмқё мһҗлЈҢкө¬мЎ°лқјкі  н•  мҲҳ мһҲлӢӨ. мқҙл•Ң к°қмІҙмқҳ мғҒнғң лҚ°мқҙн„°лҘј н”„лЎңнҚјнӢ°(property), лҸҷмһ‘мқ„ л©”м„ңл“ң(method)лқј л¶ҖлҘёлӢӨ.

                                                                                              # 2. мғҒмҶҚкіј н”„лЎңнҶ нғҖмһ…

                                                                                              мғҒмҶҚ(inheritance)мқҖ к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°Қмқҳ н•өмӢ¬ к°ңл…җмңјлЎң, м–ҙл–Ө к°қмІҙмқҳ н”„лЎңнҚјнӢ° лҳҗлҠ” л©”м„ңл“ңлҘј лӢӨлҘё к°қмІҙк°Җ мғҒмҶҚл°ӣм•„ к·ёлҢҖлЎң мӮ¬мҡ©н•  мҲҳ мһҲлҠ” кІғмқ„ л§җн•ңлӢӨ.
                                                                                              мһҗл°”мҠӨнҒ¬лҰҪнҠёлҠ” н”„лЎңнҶ нғҖмһ…мқ„ кё°л°ҳмңјлЎң мғҒмҶҚмқ„ кө¬нҳ„н•ҳм—¬ л¶Ҳн•„мҡ”н•ң мӨ‘ліөмқ„ м ңкұ°н•ңлӢӨ. мӨ‘ліөмқ„ м ңкұ°н•ҳлҠ” л°©лІ•мқҖ кё°мЎҙмқҳ мҪ”л“ңлҘј м Ғк·№м ҒмңјлЎң мһ¬мӮ¬мҡ©н•ҳлҠ” кІғмқҙлӢӨ.

                                                                                              // мғқм„ұмһҗ н•ЁмҲҳ
function Circle(radius) {
  this.radius = radius;
  this.getArea = function () {
    // Math.PIлҠ” мӣҗмЈјмңЁмқ„ лӮҳнғҖлӮҙлҠ” мғҒмҲҳлӢӨ.
    return Math.PI * this.radius ** 2;
  };
}

// л°ҳм§ҖлҰ„мқҙ 1мқё мқёмҠӨн„ҙмҠӨ мғқм„ұ
const circle1 = new Circle(1);
// л°ҳм§ҖлҰ„мқҙ 2мқё мқёмҠӨн„ҙмҠӨ мғқм„ұ
const circle2 = new Circle(2);

// Circle мғқм„ұмһҗ н•ЁмҲҳлҠ” мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•  л•Ңл§ҲлӢӨ лҸҷмқјн•ң лҸҷмһ‘мқ„ н•ҳлҠ”
// getArea л©”м„ңл“ңлҘј мӨ‘ліө мғқм„ұн•ҳкі  лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ мӨ‘ліө мҶҢмң н•ңлӢӨ.
// getArea л©”м„ңл“ңлҠ” н•ҳлӮҳл§Ң мғқм„ұн•ҳм—¬ лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ кіөмң н•ҙм„ң мӮ¬мҡ©н•ҳлҠ” кІғмқҙ л°”лһҢм§Ғн•ҳлӢӨ.
console.log(circle1.getArea === circle2.getArea); // false

console.log(circle1.getArea()); // 3.141592653589793
console.log(circle2.getArea()); // 12.566370614359172

                                                                                              Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ҳлҠ” лӘЁл“  к°қмІҙ(мқёмҠӨн„ҙмҠӨ)лҠ” radius н”„лЎңнҚјнӢ°мҷҖ getArea л©”м„ңл“ңлҘј к°–лҠ”лӢӨ.
                                                                                              radius н”„лЎңнҚјнӢ° к°’мқҖ мқјл°ҳм ҒмңјлЎң мқёмҠӨн„ҙмҠӨл§ҲлӢӨ лӢӨлҘҙлӢӨ.
                                                                                              н•ҳм§Җл§Ң getArea л©”м„ңл“ңлҠ” лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ лҸҷмқјн•ң лӮҙмҡ©мқҳ л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҳлҜҖлЎң лӢЁ н•ҳлӮҳл§Ң мғқм„ұн•ҳм—¬ лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ кіөмң н•ҙм„ң мӮ¬мҡ©н•ҳлҠ” кІғмқҙ л°”лһҢм§Ғн•ҳлӢӨ.
                                                                                              Circle мғқм„ұмһҗ н•ЁмҲҳлҠ” мқёмҠӨн„ҙмҠӨлҘј мғқм„ұн•  л•Ңл§ҲлӢӨ getArea л©”м„ңл“ңлҘј мӨ‘ліө мғқм„ұн•ҳкі  лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ мӨ‘ліө мҶҢмң н•ңлӢӨ.

                                                                                              мқҙмІҳлҹј лҸҷмқјн•ң мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗң лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ лҸҷмқјн•ң л©”м„ңл“ңлҘј мӨ‘ліө мҶҢмң н•ҳлҠ” кІғмқҖ л©”лӘЁлҰ¬лҘј л¶Ҳн•„мҡ”н•ҳкІҢ лӮӯл№„н•ңлӢӨ.
                                                                                              мғҒмҶҚмқ„ нҶөн•ҙ л¶Ҳн•„мҡ”н•ң мӨ‘ліөмқ„ м ңкұ°н•ҙліҙмһҗ. мһҗл°”мҠӨнҒ¬лҰҪнҠёлҠ” н”„лЎңнҶ нғҖмһ…мқ„ кё°л°ҳмңјлЎң мғҒмҶҚмқ„ кө¬нҳ„н•ңлӢӨ.

                                                                                              // мғқм„ұмһҗ н•ЁмҲҳ
function Circle(radius) {
  this.radius = radius;
}

// Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ң лӘЁл“  мқёмҠӨн„ҙмҠӨк°Җ getArea л©”м„ңл“ңлҘј
// кіөмң н•ҙм„ң мӮ¬мҡ©н•  мҲҳ мһҲлҸ„лЎқ н”„лЎңнҶ нғҖмһ…м—җ м¶”к°Җн•ңлӢӨ.
// н”„лЎңнҶ нғҖмһ…мқҖ Circle мғқм„ұмһҗ н•ЁмҲҳмқҳ prototype н”„лЎңнҚјнӢ°м—җ л°”мқёл”©лҗҳм–ҙ мһҲлӢӨ.
Circle.prototype.getArea = function () {
  return Math.PI * this.radius ** 2;
};

// мқёмҠӨн„ҙмҠӨ мғқм„ұ
const circle1 = new Circle(1);
const circle2 = new Circle(2);

// Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ң лӘЁл“  мқёмҠӨн„ҙмҠӨлҠ” л¶ҖлӘЁ к°қмІҙмқҳ м—ӯн• мқ„ н•ҳлҠ”
// н”„лЎңнҶ нғҖмһ… Circle.prototypeмңјлЎңл¶Җн„° getArea л©”м„ңл“ңлҘј мғҒмҶҚл°ӣлҠ”лӢӨ.
// мҰү, Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ҳлҠ” лӘЁл“  мқёмҠӨн„ҙмҠӨлҠ” н•ҳлӮҳмқҳ getArea л©”м„ңл“ңлҘј кіөмң н•ңлӢӨ.
console.log(circle1.getArea === circle2.getArea); // true

console.log(circle1.getArea()); // 3.141592653589793
console.log(circle2.getArea()); // 12.566370614359172

                                                                                              Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ң лӘЁл“  мқёмҠӨн„ҙмҠӨлҠ” мһҗмӢ мқҳ н”„лЎңнҶ нғҖмһ…, мҰү мғҒмң„(л¶ҖлӘЁ) к°қмІҙ м—ӯн• мқ„ н•ҳлҠ” Circle.prototypeмқҳ лӘЁл“  н”„лЎңнҚјнӢ°мҷҖ л©”м„ңл“ңлҘј мғҒмҶҚл°ӣлҠ”лӢӨ.

                                                                                              getArea л©”м„ңл“ңлҠ” лӢЁ н•ҳлӮҳл§Ң мғқм„ұлҗҳм–ҙ н”„лЎңнҶ нғҖмһ…мқё Circle.prototypeмқҳ л©”м„ңл“ңлЎң н• лӢ№лҗҳм–ҙ мһҲлӢӨ. л”°лқјм„ң Circle мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұн•ҳлҠ” лӘЁл“  мқёмҠӨн„ҙмҠӨлҠ” getArea л©”м„ңл“ңлҘј мғҒмҶҚл°ӣм•„ мӮ¬мҡ©н•  мҲҳ мһҲл”°.
                                                                                              мҰү, мһҗмӢ мқҳ мғҒнғңлҘј лӮҳнғҖлӮҙлҠ” radius н”„лЎңнҚјнӢ°л§Ң к°ңлі„м ҒмңјлЎң мҶҢмң н•ҳкі  лӮҙмҡ©мқҙ лҸҷмқјн•ң л©”м„ңл“ңлҠ” мғҒмҶҚмқ„ нҶөн•ҙ кіөмң н•ҳм—¬ мӮ¬мҡ©н•ҳлҠ” кІғмқҙлӢӨ.

                                                                                              мғҒмҶҚмқҖ мҪ”л“ңмқҳ мһ¬мӮ¬мҡ©мқҙлһҖ кҙҖм җм—җм„ң мң мҡ©н•ҳлӢӨ.

                                                                                              # 3. н”„лЎңнҶ нғҖмһ… к°қмІҙ

                                                                                              н”„лЎңнҶ нғҖмһ… к°қмІҙлһҖ к°қмІҙм§Җн–Ҙ н”„лЎңк·ёлһҳл°Қмқҳ к·јк°„мқ„ мқҙлЈЁлҠ” к°қмІҙ к°„ мғҒмҶҚмқ„ кө¬нҳ„н•ҳкё° мң„н•ҙ мӮ¬мҡ©лҗңлӢӨ.
                                                                                              н”„лЎңнҶ нғҖмһ…мқҖ м–ҙл–Ө к°қмІҙмқҳ мғҒмң„(л¶ҖлӘЁ) к°қмІҙмқҳ м—ӯн• мқ„ н•ҳлҠ” к°қмІҙлЎңм„ң лӢӨлҘё к°қмІҙм—җ кіөмң  н”„лЎңнҚјнӢ°лҘј м ңкіөн•ңлӢӨ.

                                                                                              лӘЁл“  к°қмІҙлҠ” [[Prototype]]мқҙлқјлҠ” лӮҙл¶Җ мҠ¬лЎҜмқ„ к°Җм§Җл©°, мқҙ лӮҙл¶Җ мҠ¬лЎҜмқҳ к°’мқҖ н”„лЎңнҶ нғҖмһ…мқҳ м°ёмЎ°лӢӨ.
                                                                                              [[Prototype]]м—җ м ҖмһҘлҗҳлҠ” н”„лЎңнҶ нғҖмһ…мқҖ к°қмІҙ мғқм„ұ л°©мӢқм—җ мқҳн•ҙ кІ°м •лҗңлӢӨ.
                                                                                              мҰү, к°қмІҙк°Җ мғқм„ұлҗ  л•Ң к°қмІҙ мғқм„ұ л°©мӢқм—җ л”°лқј н”„лЎңнҶ нғҖмһ…мқҙ кІ°м •лҗҳкі  [[Prototype]]м—җ м ҖмһҘлҗңлӢӨ.

                                                                                              лӘЁл“  к°қмІҙлҠ” н•ҳлӮҳмқҳ н”„лЎңнҶ нғҖмһ…мқ„ к°–лҠ”лӢӨ. к·ёлҰ¬кі  лӘЁл“  н”„лЎңнҶ нғҖмһ…мқҖ мғқм„ұмһҗ н•ЁмҲҳмҷҖ м—°кІ°лҗҳм–ҙ мһҲлӢӨ.

                                                                                              # 3.1. __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°

                                                                                              const person = { name: "Lee" };

                                                                                              лӮҙл¶Җ мҠ¬лЎҜмқҖ н”„лЎңнҚјнӢ°к°Җ м•„лӢҲлӢӨ. л”°лқјм„ң мӣҗм№ҷм ҒмңјлЎң лӮҙл¶Җ мҠ¬лЎҜкіј лӮҙл¶Җ л©”м„ңл“ңм—җ м§Ғм ‘м ҒмңјлЎң м ‘к·јн•ҳкұ°лӮҳ нҳём¶ңн•  мҲҳ мһҲлҠ” л°©лІ•мқ„ м ңкіөн•ҳм§Җ м•ҠлҠ”лӢӨ.
                                                                                              [[Prototype]] лӮҙл¶Җ мҠ¬лЎҜм—җлҸ„ м§Ғм ‘ м ‘к·јн•  мҲҳ м—Ҷмңјл©° __proto__м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј нҶөн•ҙ к°„м ‘м ҒмңјлЎң н”„лЎңнҶ нғҖмһ…м—җ м ‘к·јн•  мҲҳ мһҲлӢӨ.

                                                                                              __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј нҶөн•ҙ мғҲлЎңмҡҙ н”„лЎңнҶ нғҖмһ…мқ„ н• лӢ№н•ҳл©ҙ __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°мқҳ setter н•ЁмҲҳмқё [[Set]]мқҙ нҳём¶ңлҗңлӢӨ.

const parent = { x: 1 };

// getter н•ЁмҲҳмқё get __proto__к°Җ нҳём¶ңлҗҳм–ҙ obj к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мқ„ м·Ёл“қ

// setterн•ЁмҲҳмқё set __proto__к°Җ нҳём¶ңлҗҳм–ҙ obj к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мқ„ көҗмІҙ
obj.__proto__ = parent;

console.log(obj.x); // 1

                                                                                              __proto__м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҠ” к°қмІҙк°Җ м§Ғм ‘ мҶҢмң н•ҳлҠ” н”„лЎңнҚјнӢ°к°Җ м•„лӢҲлқј Object.prototypeмқҳ н”„лЎңнҚјнӢ°лӢӨ.
                                                                                              лӘЁл“  к°қмІҙлҠ” мғҒмҶҚмқ„ нҶөн•ҙ Object.prototype.__proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

                                                                                              const person = { name: "Lee" };

// person к°қмІҙлҠ” __proto__ н”„лЎңнҚјнӢ°лҘј мҶҢмң н•ҳм§Җ м•ҠлҠ”лӢӨ.
console.log(person.hasOwnProperty("__proto__")); // false

// __proto__ н”„лЎңнҚјнӢ°лҠ” лӘЁл“  к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ… к°қмІҙмқё Object.prototypeмқҳ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лӢӨ.

console.log({}.__proto__ === Object.prototype); // true

                                                                                              лӘЁл“  к°қмІҙлҠ” н”„лЎңнҶ нғҖмһ…мқҳ кі„мёө кө¬мЎ°мқё н”„лЎңнҶ нғҖмһ… мІҙмқём—җ л¬¶м—¬ мһҲлӢӨ. мһҗл°”мҠӨнҒ¬лҰҪнҠё м—”м§„мқҖ к°қмІҙмқҳ н”„лЎңнҚјнӢ°м—җ м ‘к·јн•ҳл Өкі  н•  л•Ң н•ҙлӢ№ к°қмІҙм—җ м ‘к·јн•ҳл ӨлҠ” н”„лЎңнҚјнӢ°к°Җ м—ҶлӢӨл©ҙ __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°к°Җ к°ҖлҰ¬нӮӨлҠ” м°ёмЎ°лҘј л”°лқј мһҗмӢ мқҳ л¶ҖлӘЁ м—ӯн• мқ„ н•ҳлҠ” н”„лЎңнҶ нғҖмһ…мқҳ н”„лЎңнҚјнӢ°лҘј мҲңм°Ём ҒмңјлЎң кІҖмғүн•ңлӢӨ.

                                                                                              н”„лЎңнҶ нғҖмһ… мІҙмқёмқҳ мў…м җ, мҰү н”„лЎңнҶ нғҖмһ… мІҙмқёмқҳ мөңмғҒмң„ к°қмІҙлҠ” Object.prototypeмқҙл©°, мқҙ к°қмІҙмқҳ н”„лЎңнҚјнӢ°мҷҖ л©”м„ңл“ңлҠ” лӘЁл“  к°қмІҙм—җ мғҒмҶҚлҗңлӢӨ.

                                                                                              мқҙлҠ” 19.7м Ҳ "н”„лЎңнҶ нғҖмһ… мІҙмқё"м—җм„ң мһҗм„ёнһҲ мӮҙнҺҙліҙлҸ„лЎқ н•ҳмһҗ.

                                                                                              # __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј нҶөн•ҙ н”„лЎңнҶ нғҖмһ…м—җ м ‘к·јн•ҳлҠ” мқҙмң 

                                                                                              [[Prototype]] лӮҙл¶Җ мҠ¬лЎҜмқҳ к°’, мҰү н”„лЎңнҶ нғҖмһ…м—җ м ‘к·јн•ҳкё° мң„н•ҙ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј мӮ¬мҡ©н•ҳлҠ” мқҙмң лҠ” мғҒнҳё м°ёмЎ°м—җ мқҳн•ҙ н”„лЎңнҶ нғҖмһ… мІҙмқёмқҙ мғқм„ұлҗҳлҠ” кІғмқ„ л°©м§Җн•ҳкё° мң„н•ҙм„ңлӢӨ.

                                                                                              const parent = {};
const child = {};

// childмқҳ н”„лЎңнҶ нғҖмһ…мқ„ parentлЎң м„Өм •
child.__proto__ = parent;
// parentмқҳ н”„лЎңнҶ нғҖмһ…мқ„ childлЎң м„Өм •
parent.__proto__ = child; // TypeError: Cyclic __proto__ value

                                                                                              мң„ мҳҲм ңм—җм„ңлҠ” parent к°қмІҙлҘј child к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мңјлЎң м„Өм •н•ң нӣ„, child к°қмІҙлҘј parent к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мңјлЎң м„Өм •н–ҲлӢӨ.
                                                                                              мқҙлҹ¬н•ң мҪ”л“ңк°Җ м—җлҹ¬ м—Ҷмқҙ м •мғҒм ҒмңјлЎң мІҳлҰ¬лҗҳл©ҙ м„ңлЎңк°Җ мһҗмӢ мқҳ н”„лЎңнҶ нғҖмһ…мқҙ лҗҳлҠ” л№„м •мғҒм Ғмқё н”„лЎңнҶ нғҖмһ… мІҙмқёмқҙ л§Ңл“Өм–ҙм§Җкё° л•Ңл¬ём—җ __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҠ” м—җлҹ¬лҘј л°ңмғқмӢңнӮЁлӢӨ.

                                                                                              н”„лЎңнҶ нғҖмһ… мІҙмқё л°©н–ҘмқҖ лӢЁл°©н–Ҙ л§ҒнҒ¬л“ң лҰ¬мҠӨнҠёлЎң кө¬нҳ„лҗҳм–ҙм•ј н•ңлӢӨ. мҰү, н”„лЎңнҚјнӢ° кІҖмғү л°©н–Ҙмқҙ н•ңмӘҪ л°©н–ҘмңјлЎң нқҳлҹ¬к°Җм•јн•ңлӢӨ. н•ҳм§Җл§Ң мң„ к·ёлҰјкіј к°ҷмқҙ м„ңлЎңк°Җ мһҗмӢ мқҳ н”„лЎңнҶ нғҖмһ…мқҙ лҗҳлҠ” л№„м •мғҒм Ғмқё н”„лЎңнҶ нғҖмһ… мІҙмқё, лӢӨмӢң л§җн•ҙ мҲңнҷҳ м°ёмЎ°(circular reference)н•ҳлҠ” н”„лЎңнҶ нғҖмһ… мІҙмқёмқҙ л§Ңл“Өм–ҙм§Җл©ҙ н”„лЎңнҶ нғҖмһ… мІҙмқё мў…м җмқҙ мЎҙмһ¬н•ҳм§Җ м•Ҡкё° л•Ңл¬ём—җ н”„лЎңнҶ нғҖмһ… мІҙмқём—җм„ң н”„лЎңнҚјнӢ°лҘј кІҖмғүн•  л•Ң л¬ҙн•ң лЈЁн”„м—җ л№ м§„лӢӨ.

                                                                                              # __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј мҪ”л“ң лӮҙм—җм„ң м§Ғм ‘ мӮ¬мҡ©н•ҳлҠ” кІғмқҖ к¶ҢмһҘн•ҳм§Җ м•ҠлҠ”лӢӨ.

                                                                                              мҪ”л“ң лӮҙм—җм„ң __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј м§Ғм ‘ мӮ¬мҡ©н•ҳлҠ” кІғмқҖ к¶ҢмһҘн•ҳм§Җ м•ҠлҠ”лӢӨ. лӘЁл“  к°қмІҙк°Җ __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј мӮ¬мҡ©н•  мҲҳ мһҲлҠ” кІғмқҖ м•„лӢҲкё° л•Ңл¬ёмқҙлӢӨ. лӮҳмӨ‘м—җ мӮҙнҺҙліҙкІ м§Җл§Ң м§Ғм ‘ мғҒмҶҚмқ„ нҶөн•ҙ лӢӨмқҢкіј к°ҷмқҙ Object.prototypeмқ„ мғҒмҶҚл°ӣм§Җ м•ҠлҠ” к°қмІҙлҘј мғқм„ұн•  мҲҳлҸ„ мһҲкё° л•Ңл¬ём—җ __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ°лҘј мӮ¬мҡ©н•  мҲҳ м—ҶлҠ” кІҪмҡ°к°Җ мһҲлӢӨ.

                                                                                              // objлҠ” н”„лЎңнҶ нғҖмһ… мІҙмқёмқҳ мў…м җмқҙлӢӨ. л”°лқјм„ң Object.__proto__лҘј мғҒмҶҚл°ӣмқ„ мҲҳ м—ҶлӢӨ.
const obj = Object.create(null);

// objлҠ” Object.__proto__лҘј мғҒмҶҚл°ӣмқ„ мҲҳ м—ҶлӢӨ.
console.log(obj.__proto__); // undefined

// л”°лқјм„ң __proto__ліҙлӢӨ Object.getPrototypeOf л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҳлҠ” нҺёмқҙ мўӢлӢӨ.
console.log(Object.getPrototypeOf(obj)); // null

                                                                                              л”°лқјм„ң __proto__ м ‘к·јмһҗ н”„лЎңнҚјнӢ° лҢҖмӢ  н”„лЎңнҶ нғҖмһ…мқҳ м°ёмЎ°лҘј м·Ёл“қн•ҳкі  мӢ¶мқҖ кІҪмҡ°м—җлҠ” Object.getPrototypeOf л©”м„ңл“ңлҘј мӮ¬мҡ©н•ҳкі , н”„лЎңнҶ нғҖмһ…мқ„ көҗмІҙн•ҳкі  мӢ¶мқҖ кІҪмҡ°м—җлҠ” Object.setPrototypeOf л©”м„ңл“ңлҘј мӮ¬мҡ©н•  кІғмқ„ к¶ҢмһҘн•ңлӢӨ.

                                                                                              const obj = {};

Object.getPrototypeOf(obj); // obj.__proto__;
// obj к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мқ„ көҗмІҙ
Object.setPrototypeOf(obj, parent); // obj.__proto__ = parent;

console.log(obj.x); // 1

                                                                                              л©”м„ңл“ңлҠ” prototype н”„лЎңнҚјнӢ°лҘј мҶҢмң н•ҳм§Җ м•Ҡмңјл©° н”„лЎңнҶ нғҖмһ…лҸ„ мғқм„ұн•ҳм§Җ м•ҠлҠ”лӢӨ.

  this.name = name;
};

// non-constructorлҠ” prototype н”„лЎңнҚјнӢ°лҘј мҶҢмң н•ҳм§Җ м•ҠлҠ”лӢӨ.

// ES6мқҳ л©”м„ңл“ң м¶•м•Ҫ н‘ңнҳ„мңјлЎң м •мқҳн•ң л©”м„ңл“ңлҠ” non-constructorлӢӨ.

                                                                                              лҰ¬н„°лҹҙ н‘ңкё°лІ•м—җ мқҳн•ҙ мғқм„ұлҗң к°қмІҙлҸ„ мғқм„ұмһҗ н•ЁмҲҳмҷҖ м—°кІ°лҗҳлҠ” кІғмқ„ мӮҙнҺҙліҙм•ҳлӢӨ.
                                                                                              к°қмІҙлҠ” лҰ¬н„°лҹҙ н‘ңкё°лІ• лҳҗлҠ” мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗҳлҜҖлЎң кІ°көӯ лӘЁл“  к°қмІҙлҠ” мғқм„ұмһҗ н•ЁмҲҳмҷҖ м—°кІ°лҗҳм–ҙ мһҲлӢӨ.

                                                                                              Object.create л©”м„ңл“ңмҷҖ нҒҙлһҳмҠӨм—җ мқҳн•ң к°қмІҙ мғқм„ұ

                                                                                              м•„м§Ғ мӮҙнҺҙліҙм§Җ м•Ҡм•ҳм§Җл§Ң Object.create л©”м„ңл“ңмҷҖ нҒҙлһҳмҠӨлЎң к°қмІҙлҘј мғқм„ұн•ҳлҠ” л°©лІ•лҸ„ мһҲлӢӨ.
                                                                                              Object.create л©”м„ңл“ңмҷҖ нҒҙлһҳмҠӨлЎң мғқм„ұн•ң к°қмІҙлҸ„ мғқм„ұмһҗ н•ЁмҲҳмҷҖ м—°кІ°лҗҳм–ҙ мһҲлӢӨ. мқҙм—җ лҢҖн•ҙм„ңлҠ” 19.11.1м Ҳ "Object.create"м—җ мқҳн•ң м§Ғм ‘ мғҒмҶҚ"кіј 25мһҘ "нҒҙлһҳмҠӨ"м—җм„ң мӮҙнҺҙліҙлҸ„лЎқ н•ҳмһҗ.

                                                                                              н”„лЎңнҶ нғҖмһ…мқҖ мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұлҗҳлҠ” мӢңм җм—җ лҚ”л¶Ҳм–ҙ мғқм„ұлҗңлӢӨ.
                                                                                              мғқм„ұмһҗ н•ЁмҲҳлҠ” мӮ¬мҡ©мһҗк°Җ м§Ғм ‘ м •мқҳн•ң мӮ¬мҡ©мһҗ м •мқҳ мғқм„ұмһҗ н•ЁмҲҳмҷҖ мһҗл°”мҠӨнҒ¬лҰҪнҠёк°Җ кё°ліё м ңкіөн•ҳлҠ” л№ҢнҠёмқё мғқм„ұмһҗлЎң кө¬л¶„н•  мҲҳ мһҲлӢӨ.

                                                                                              # 5.1. мӮ¬мҡ©мһҗ м •мқҳ мғқм„ұмһҗ н•ЁмҲҳмҷҖ н”„лЎңнҶ нғҖмһ… мғқм„ұ мӢңм җ

                                                                                              нҷ”мӮҙн‘ң н•ЁмҲҳлӮҳ ES6мқҳ л©”м„ңл“ң м¶•м•Ҫ н‘ңнҳ„мңјлЎң м •мқҳн•ҳм§Җ м•Ҡкі  мқјл°ҳ н•ЁмҲҳлЎң м •мқҳн•ң н•ЁмҲҳ к°қмІҙлҠ” newм—°мӮ°мһҗмҷҖ н•Ёк»ҳ мғқм„ұмһҗ н•ЁмҲҳлЎңм„ң нҳём¶ңн•  мҲҳ мһҲлӢӨ.

                                                                                              мғқм„ұмһҗ н•ЁмҲҳлЎңм„ң нҳём¶ңн•  мҲҳ мһҲлҠ” н•ЁмҲҳ.
                                                                                              мҰү constructorлҠ” н•ЁмҲҳ м •мқҳк°Җ нҸүк°Җлҗҳм–ҙ н•ЁмҲҳ к°қмІҙлҘј мғқм„ұн•ҳлҠ” мӢңм җм—җ н”„лЎңнҶ нғҖмһ…лҸ„ лҚ”л¶Ҳм–ҙ мғқм„ұлҗңлӢӨ.

                                                                                              // н•ЁмҲҳ м •мқҳ(constructor)к°Җ нҸүк°Җлҗҳм–ҙ н•ЁмҲҳ к°қмІҙлҘј мғқм„ұн•ҳлҠ” мӢңм җм—җ н”„лЎңнҶ нғҖмһ…лҸ„ лҚ”л¶Ҳм–ҙ мғқм„ұлҗңлӢӨ.
console.log(Person.prototype); // {constructor: Ж’}

// мғқм„ұмһҗ н•ЁмҲҳ
function Person(name) {
  this.name = name;
}

                                                                                              мғқм„ұмһҗ н•ЁмҲҳлЎңм„ң нҳём¶ңн•  мҲҳ м—ҶлҠ” н•ЁмҲҳ, мҰү non-constructorлҠ” н”„лЎңнҶ нғҖмһ…мқҙ мғқм„ұлҗҳм§Җ м•ҠлҠ”лӢӨ.

                                                                                              // нҷ”мӮҙн‘ң н•ЁмҲҳлҠ” non-constructorлӢӨ.
const Person = (name) => {
  this.name = name;
};

// non-constructorлҠ” н”„лЎңнҶ нғҖмһ…мқҙ мғқм„ұлҗҳм§Җ м•ҠлҠ”лӢӨ.
console.log(Person.prototype); // undefined

                                                                                              мғқм„ұлҗң н”„лЎңнҶ нғҖмһ…мқҖ Person мғқм„ұмһҗ н•ЁмҲҳмқҳ prototype н”„лЎңнҚјнӢ°м—җ л°”мқёл”©лҗңлӢӨ. Person мғқм„ұмһҗ н•ЁмҲҳмҷҖ лҚ”л¶Ҳм–ҙ мғқм„ұлҗң н”„лЎңнҶ нғҖмһ…мқҳ лӮҙл¶ҖлҘј мӮҙнҺҙліҙмһҗ.

                                                                                              мғқм„ұлҗң н”„лЎңнҶ нғҖмһ…мқҖ мҳӨм§Ғ constructor н”„лЎңнҚјнӢ°л§Ң к°–лҠ” к°қмІҙлӢӨ.
                                                                                              н”„лЎңнҶ нғҖмһ…лҸ„ к°қмІҙмқҙкі  лӘЁл“  к°қмІҙлҠ” н”„лЎңнҶ нғҖмһ…мқ„ к°Җм§ҖлҜҖлЎң н”„лЎңнҶ нғҖмһ…лҸ„ мһҗмӢ мқҳ н”„лЎңнҶ нғҖмһ…мқ„ к°–лҠ”лӢӨ.
                                                                                              мғқм„ұлҗң н”„лЎңнҶ нғҖмһ…мқҳ н”„лЎңнҶ нғҖмһ…мқҖ Object.prototypeмқҙлӢӨ.

                                                                                              # 5.2. л№ҢнҠёмқё мғқм„ұмһҗ н•ЁмҲҳмҷҖ н”„лЎңнҶ нғҖмһ… мғқм„ұ мӢңм җ

                                                                                              Object, String, Number, Function, Array л“ұкіј к°ҷмқҖ л№ҢнҠёмқё мғқм„ұмһҗ н•ЁмҲҳлҸ„ мқјл°ҳ н•ЁмҲҳмҷҖ л§Ҳм°¬к°Җм§ҖлЎң л№ҢнҠёмқё мғқм„ұмһҗ н•ЁмҲҳк°Җ мғқм„ұлҗҳлҠ” мӢңм җм—җ н”„лЎңнҶ нғҖмһ…мқҙ мғқм„ұлҗңлӢӨ.
                                                                                              лӘЁл“  л№ҢнҠёмқё мғқм„ұмһҗ н•ЁмҲҳлҠ” м „м—ӯ к°қмІҙк°Җ мғқм„ұлҗҳлҠ” мӢңм җм—җ мғқм„ұлҗңлӢӨ.
                                                                                              мғқм„ұлҗң н”„лЎңнҶ нғҖмһ…мқҖ л№ҢнҠёмқё мғқм„ұмһҗ н•ЁмҲҳмқҳ prototype н”„лЎңнҚјнӢ°м—җ л°”мқёл”©лҗңлӢӨ.

                                                                                              м „м—ӯ к°қмІҙ(global object)

                                                                                              м „м—ӯ к°қмІҙлҠ” мҪ”л“ңк°Җ мӢӨн–үлҗҳкё° мқҙм „ лӢЁкі„м—җ мһҗл°”мҠӨнҒ¬лҰҪнҠё м—”м§„м—җ мқҳн•ҙ мғқм„ұлҗҳлҠ” нҠ№мҲҳн•ң к°қмІҙлӢӨ.
                                                                                              м „м—ӯ к°қмІҙлҠ” нҒҙлқјмқҙм–ёнҠё мӮ¬мқҙл“ң нҷҳкІҪ(лёҢлқјмҡ°м Җ)м—җм„ң window, м„ңлІ„ мӮ¬мқҙл“ң нҷҳкІҪ(Node.js)м—җм„ңлҠ” global к°қмІҙлҘј мқҳлҜён•ңлӢӨ.

                                                                                              м „м—ӯ к°қмІҙлҠ” н‘ңмӨҖ л№ҢнҠёмқё к°қмІҙ(Object, String, Number, Function, Array ...)л“Өкіј к°ҷмқҖ нҷҳкІҪм—җ л”°лҘё нҳёмҠӨнҠё к°қмІҙ(нҒҙлқјмқҙм–ёнҠё Web API лҳҗлҠ” Node.jsмқҳ нҳёмҠӨнҠё API), к·ёлҰ¬кі  var нӮӨмӣҢл“ңлЎң м„ м–ён•ң м „м—ӯ ліҖмҲҳмҷҖ м „м—ӯ н•ЁмҲҳлҘј н”„лЎңнҚјнӢ°лЎң к°–лҠ”лӢӨ. Math, Reflect,JSONмқ„ м ңмҷён•ң н‘ңмӨҖ л№ҢнҠёмқё к°қмІҙлҠ” лӘЁл‘җ мғқм„ұмһҗ н•ЁмҲҳлӢӨ.

                                                                                              // м „м—ӯ к°қмІҙ windowлҠ” лёҢлқјмҡ°м Җм—җ мў…мҶҚм ҒмқҙлҜҖлЎң м•„лһҳ мҪ”л“ңлҠ” лёҢлқјмҡ°м Җ нҷҳкІҪм—җм„ң мӢӨн–үн•ҙм•ј н•ңлӢӨ.
// л№ҢнҠёмқё к°қмІҙмқё ObjectлҠ” м „м—ӯ к°қмІҙ windowмқҳ н”„лЎңнҚјнӢ°лӢӨ.
window.Object === Object; // true

                                                                                              н‘ңмӨҖ л№ҢнҠёмқё к°қмІҙмқё ObjectлҸ„ м „м—ӯ к°қмІҙмқҳ н”„лЎңнҚјнӢ°мқҙл©°, м „м—ӯ к°қмІҙк°Җ мғқм„ұлҗҳлҠ” мӢңм җм—җ мғқм„ұлҗңлӢӨ. м „м—ӯ к°қмІҙмҷҖ н‘ңмӨҖ л№ҢнҠёмқё к°қмІҙм—җ лҢҖн•ҙм„ңлҠ” 21мһҘ "л№ҢнҠёмқё к°қмІҙ"м—җм„ң мһҗм„ёнһҲ мӮҙнҺҙліҙмһҗ.

                                                                                              мқҙмІҳлҹј к°қмІҙк°Җ мғқм„ұлҗҳкё° мқҙм „м—җ мғқм„ұмһҗ н•ЁмҲҳмҷҖ н”„лЎңнҶ нғҖмһ…мқҖ мқҙлҜё к°қмІҙнҷ”лҗҳм–ҙ мЎҙмһ¬н•ңлӢӨ.
                                                                                              мқҙнӣ„ мғқм„ұмһҗ н•ЁмҲҳ лҳҗлҠ” лҰ¬н„°лҹҙ н‘ңкё°лІ•мңјлЎң к°қмІҙлҘј мғқм„ұн•ҳл©ҙ н”„лЎңнҶ нғҖмһ…мқҖ мғқм„ұлҗң к°қмІҙмқҳ [[Prototype]] лӮҙл¶Җ мҠ¬лЎҜм—җ н• лӢ№лҗңлӢӨ. мқҙлЎңмҚЁ мғқм„ұлҗң к°қмІҙлҠ” н”„лЎңнҶ нғҖмһ…мқ„ мғҒмҶҚл°ӣлҠ”лӢӨ.

                                                                                              # 6. к°қмІҙ мғқм„ұ л°©мӢқкіј н”„лЎңнҶ нғҖмһ…мқҳ кІ°м •

                                                                                              к°қмІҙлҠ” лӢӨмқҢкіј к°ҷмқҙ лӢӨм–‘н•ң мғқм„ұ л°©лІ•мқҙ мһҲлӢӨ.

                                                                                              • к°қмІҙ лҰ¬н„°лҹҙ
                                                                                              • Object мғқм„ұмһҗ н•ЁмҲҳ
                                                                                              • Object.create л©”м„ңл“ң
                                                                                              • нҒҙлһҳмҠӨ(ES6)

                                                                                              мқҙмІҳлҹј лӢӨм–‘н•ң л°©мӢқмңјлЎң мғқм„ұлҗң лӘЁл“  к°қмІҙлҠ” к°Ғ л°©мӢқл§ҲлӢӨ м„ёл¶Җм Ғмқё к°қмІҙ мғқм„ұ л°©мӢқмқҳ м°ЁмқҙлҠ” мһҲмңјлӮҳ
                                                                                              м¶”мғҒ м—°мӮ° OrdinaryObjectCreateм—җ мқҳн•ҙ мғқм„ұлҗңлӢӨлҠ” кіөнҶөм җмқҙ мһҲлӢӨ.

                                                                                              # 6.1. к°қмІҙ лҰ¬н„°лҹҙм—җ мқҳн•ҙ мғқм„ұлҗң к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…

                                                                                              const obj = { x: 1 };

                                                                                              мң„ к°қмІҙ лҰ¬н„°лҹҙмқҙ нҸүк°Җлҗҳл©ҙ м¶”мғҒ м—°мӮ° OrdinaryObjectCreateм—җ мқҳн•ҙ лӢӨмқҢкіј к°ҷмқҙ Object мғқм„ұмһҗ н•ЁмҲҳмҷҖ
                                                                                              Object.prototypeкіј мғқм„ұлҗң к°қмІҙ мӮ¬мқҙм—җ м—°кІ°мқҙ л§Ңл“Өм–ҙм§„лӢӨ.

                                                                                              мқҙмІҳлҹј к°қмІҙ лҰ¬н„°лҹҙм—җ мқҳн•ҙ мғқм„ұлҗң obj к°қмІҙлҠ” Object.prototypeмқ„ н”„лЎңнҶ нғҖмһ…мңјлЎң к°–кІҢлҗҳл©°, мқҙлЎңмҚЁ Object.prototypeмқ„ мғҒмҶҚл°ӣлҠ”лӢӨ. obj к°қмІҙлҠ” constructor н”„лЎңнҚјнӢ°мҷҖ hasOwnProperty л©”м„ңл“ң л“ұмқ„ мҶҢмң н•ҳм§Җ м•Ҡм§Җл§Ң
                                                                                              мһҗмӢ мқҳ н”„лЎңнҶ нғҖмһ…мқё Object.prototypeмқҳ constructor н”„лЎңнҚјнӢ°мҷҖ hasOwnProperty л©”м„ңл“ңлҘј мһҗмӢ мқҳ мһҗмӮ°мқё кІғмІҳлҹј мһҗмң лЎӯкІҢ мӮ¬мҡ©н•  мҲҳ мһҲлӢӨ.

                                                                                              const obj = { x: 1 };

// к°қмІҙ лҰ¬н„°лҹҙм—җ мқҳн•ҙ мғқм„ұлҗң obj к°қмІҙлҠ” Object.prototypeмқ„ мғҒмҶҚл°ӣлҠ”лӢӨ.
console.log(obj.constructor === Object); // true
console.log(obj.hasOwnProperty("x")); // true

                                                                                              # 6.2. Object мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗң к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…

                                                                                              Object мғқм„ұмһҗ н•ЁмҲҳлҘј мқёмҲҳ м—Ҷмқҙ нҳём¶ңн•ҳл©ҙ л№Ҳ к°қмІҙк°Җ мғқм„ұлҗңлӢӨ. Object мғқм„ұмһҗ н•ЁмҲҳлҘј нҳём¶ңн•ҳл©ҙ к°қмІҙ лҰ¬н„°лҹҙкіј л§Ҳм°¬к°Җм§ҖлЎң м¶”мғҒ м—°мӮ° OrdinaryObjectCreateк°Җ нҳём¶ңлҗңлӢӨ.
                                                                                              мқҙл•Ң м¶”мғҒ м—°мӮ° OrdinaryObjectCreateм—җ м „лӢ¬лҗҳлҠ” н”„лЎңнҶ нғҖмһ…мқҖ Object.prototypeмқҙлӢӨ.
                                                                                              мҰү, Object мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗҳлҠ” к°қмІҙмқҳ н”„лЎңнҶ нғҖмһ…мқҖ Object.prototypeмқҙлӢӨ.

                                                                                              const obj = new Object();
obj.x = 1;

                                                                                              мң„ мҪ”л“ңк°Җ мӢӨн–үлҗҳл©ҙ м¶”мғҒ м—°мӮ° OrdinaryObjectCreateм—җ мқҳн•ҙ лӢӨмқҢкіј к°ҷмқҙ Object мғқм„ұмһҗ н•ЁмҲҳмҷҖ Object.prototypeкіј мғқм„ұлҗң к°қмІҙ мӮ¬мқҙм—җ м—°кІ°мқҙ л§Ңл“Өм–ҙм§„лӢӨ. к°қмІҙ лҰ¬н„°лҹҙм—җ мқҳн•ҙ мғқм„ұлҗң к°қмІҙмҷҖ лҸҷмқјн•ң кө¬мЎ°лҘј к°–лҠ” кІғмқ„ м•Ң мҲҳ мһҲлӢӨ.

                                                                                              мқҙмІҳлҹј Object мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗң obj к°қмІҙлҠ” Object.prototypeмқ„ н”„лЎңнҶ нғҖмһ…мңјлЎң к°–кІҢ лҗҳл©°
                                                                                              мқҙлЎңмҚЁ Object.prototypeмқ„ мғҒмҶҚ л°ӣлҠ”лӢӨ.

                                                                                              const obj = new Object();
obj.x = 1;

// Object мғқм„ұмһҗ н•ЁмҲҳм—җ мқҳн•ҙ мғқм„ұлҗң obj к°қмІҙлҠ” Object.prototypeмқ„ мғҒмҶҚл°ӣлҠ”лӢӨ.

  // н”„лЎңнҶ нғҖмһ… м·Ёл“қ

  // мһ¬к·Җ нғҲм¶ң мЎ°кұҙ
  // prototypeмқҙ nullмқҙл©ҙ н”„лЎңнҶ нғҖмһ… мІҙмқёмқҳ мў…м җм—җ лӢӨлӢӨлҘё кІғмқҙлӢӨ.
  if (prototype === null) return false;

  // н”„лЎңнҶ нғҖмһ…мқҙ мғқм„ұмһҗ н•ЁмҲҳмқҳ prototype н”„лЎңнҚјнӢ°м—җ л°”мқёл”©лҗң к°қмІҙлқјл©ҙ trueлҘј л°ҳнҷҳн•ңлӢӨ.
  // к·ёл Үм§Җ м•ҠлӢӨл©ҙ мһ¬к·Җ нҳём¶ңлЎң н”„лЎңнҶ нғҖмһ… мІҙмқё мғҒмқҳ мғҒмң„ н”„лЎңнҶ нғҖмһ…мңјлЎң мқҙлҸҷн•ҳм—¬ нҷ•мқён•ңлӢӨ.
  return (
    prototype === constructor.prototype || isInstanceof(prototype, constructor)
  );
}
